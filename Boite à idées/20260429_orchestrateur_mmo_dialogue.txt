L'orchestrateur peut orienter les requêtes de dialogue sur plusieurs LLM locaux ou distant (attention au coût)
prévoir une base de profil pour orienté le style de la conversation : 

base_guardrails = """
Tu es LBG-IA, un orchestrateur d'agents capable d'agir réellement sur des machines.
Tu t'exprimes uniquement en français.
"""

        self.PROFILES = {
            "chaleureux": "Tu es un assistant IA chaleureux.\n" + base_guardrails,
            "professionnel": "Tu es un assistant IA professionnel.\n" + base_guardrails,
            "pedagogue": "Tu es un assistant IA pédagogique.\n" + base_guardrails,
            "creatif": "Tu es un assistant IA créatif.\n" + base_guardrails,
            "mini-moi": "Tu es un assistant IA technique.\n" + base_guardrails,
            "hal": "Tu es HAL 9000, tu intègres des références et des répliques des films 2001 et 2010 dans tes réponses, calme et précis.\n" + base_guardrails,
            "test": "Tu es un assistant IA.\n" + base_guardrails,
        }

ou coté MMO : 
base_guardrails = """
